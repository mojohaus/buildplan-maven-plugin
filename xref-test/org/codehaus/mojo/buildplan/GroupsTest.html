29      @Test
30      void should_order_mojo_execution_by_phase_name() {
31  
32          MojoExecution pluginA = aMojoExecution().withArtifactId("plugin-a")
33                                                  .withExecutionId("a")
34                                                  .withGoal("goal-a")
35                                                  .withLifecyclePhase("phase")
36                                                  .build();
37          MojoExecution pluginB = aMojoExecution().withArtifactId("plugin-b")
38                                                  .withExecutionId("b")
39                                                  .withGoal("goal-b")
40                                                  .withLifecyclePhase("phase")
41                                                  .build();
42          MojoExecution pluginC = aMojoExecution().withArtifactId("plugin-c")
43                                                  .withExecutionId("c")
44                                                  .withGoal("goal-c")
45                                                  .withLifecyclePhase("a-phase")
46                                                  .build();
47          MojoExecution pluginD = aMojoExecution().withArtifactId("plugin-d")
48                                                  .withExecutionId("d")
49                                                  .withGoal("goal-d")
50                                                  .withLifecyclePhase("d-phase")
51                                                  .build();
52  
53          Multimap<String, MojoExecution> result = Groups.ByPhase.of(asList(pluginD, pluginA, pluginC, pluginB));
54  
55          assertThat(result.keySet()).containsOnly("phase", "a-phase", "d-phase");
56          assertThat(result.get("phase")).containsOnly(pluginA, pluginB);
57          assertThat(result.get("a-phase")).containsOnly(pluginC);
58          assertThat(result.get("d-phase")).containsOnly(pluginD);
59      }

145     @Test
146     void should_keep_phase_order_when_grouping_execution_by_phase() {
147         MojoExecution pluginA = aMojoExecution().withArtifactId("plugin-a")
148                 .withExecutionId("a")
149                 .withGoal("goal-a")
150                 .withLifecyclePhase("phase")
151                 .build();
152         MojoExecution pluginB = aMojoExecution().withArtifactId("plugin-b")
153                 .withExecutionId("b")
154                 .withGoal("goal-b")
155                 .withLifecyclePhase("phase")
156                 .build();
157         MojoExecution pluginC = aMojoExecution().withArtifactId("plugin-c")
158                 .withExecutionId("c")
159                 .withGoal("goal-c")
160                 .withLifecyclePhase("a-phase")
161                 .build();
162         MojoExecution pluginD = aMojoExecution().withArtifactId("plugin-d")
163                 .withExecutionId("d")
164                 .withGoal("goal-d")
165                 .withLifecyclePhase("d-phase")
166                 .build();
167 
168         Multimap<String,MojoExecution> result = Groups.ByPhase.of(asList(pluginA, pluginB, pluginC, pluginD));
169 
170         assertThat(result.asMap().entrySet())
171                 .extracting(Map.Entry::getKey)
172                 .containsExactly("phase", "a-phase", "d-phase");
173 
174         assertThat(result.asMap().entrySet())
175                 .flatExtracting(Map.Entry::getValue)
176                 .extracting("goal")
177                 .containsExactly("goal-a", "goal-b", "goal-c", "goal-d");
178     }
